\begin{theorem}
  Let $S, T \in \LT(V)$ commute and $\lambda \in F$.
  Show that $E(\lambda, S)$ is invariant under $T$.
\end{theorem}
\begin{proof}
  For any $v \in E(\lambda, S)$,
  $STv = TSv = T \lambda v = \lambda Tv$, thus $Tv \in E(\lambda, S)$.
\end{proof}
